procedure trouverCycles (G = (S, A))
	Vu : tableau de 1 à nbSommets, init à Faux
	CyclesTrouves : tableau de cycles, init à vide
	v : sommet
	
	tant que Vu n'est pas tout vrai
		v := premier sommet non vu
		Vu[v] := vrai
		CyclesTrouves.ajouter(TrouverCyclesRec(G, v, Explore init à faux, Pile de sommets vide, Vu))
	fintq

	retourne CyclesTrouves
fin

procedure trouverCyclesRec (G = (S, A), v : Sommet, Explore : tableau de 1 à nbSommets init à faux, SommetsParcourus : Pile de sommets ,Vu)
	
	si Explore[v] = Vrai
		c := construire le cycle avec la Pile
		ajouterCycle(c)
	sinon
		pour tous les voisins voisin de v
			Explore[v] = Vrai
			TrouverCyclesRec(voisin, Explore, SommetsParcourus, Vu)
		fin pour
	finsi
fin
