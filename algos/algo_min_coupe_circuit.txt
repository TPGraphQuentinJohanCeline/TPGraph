fonction MinCoupeCircuit(k : entier, G = (S, A)) retourne entier (0 si il y a une solution)
	Cycles : tableau vide de cycles
	Cycles := trouverCycles(G)
	donnerPoidsAuxArrêtes(G, Cycles) // parcourt les cycles et incremente le poids des arêtes

	pour toutes arêtes de poids non nul
		supprimer arête
		si plus de cycles
			retourne 0
		sinon si MinCoupeCircuitRec(k-1, G, Cycles)
			retourne 0
		sinon
			retourne 1
		finsi
	finpour
fin

fonction MinCoupeCircuitRec(k:entier, G = (S,A), Cycles : tableau des cycles contenus dans le graphe) retourne boolean
	si k = 0
		retourne Faux
	finsi

	supprimer arrête de plus haut poids

	si plus de cycles
		retourne Vrai
	sinon
		pour toute arête de poids non nul
			si MinCoupeCircuitRec(k-1, G, Cycles)
				retourne Vrai
			finsi
		fin pour

		retourne Faux
	finsi
fin

procedure supprimerArête(G = (S,A), a : Arête, Cycles)
	enlève a de G
	pour tout cycle contenant a faire
		enlever a
		pour toute arête du cycle
			décrémenter poids de arête
		finpour
		supprimer cycle
	finpour
fin
